Standard for emitted stats objects:

event: constants -> EVENTS enum

Search Request:
{
    event
    id: search request id
    source_name: requesting node name
    query: search query
}


Search Relay:
{
    event
    id: search request id
    source_name: requesting node name
}


Search Response:
{
    event
    id: search request id
    source_name: requesting node name
}

Search Discard:
{
    event
    id: search request id
    source_name: requesting node name
}

Search Revisit:
{
    event
    id: search request id
    source_name: requesting node name
}

Encounter Interval:
{
    event
    sample_size
    interval_count
    total_avg
    source_name: requesting node name
}
